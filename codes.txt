# 0) Install deps (example for ROS 2 Jazzy)
sudo apt install ros-jazzy-gz-ros2-control ros-jazzy-ros2-control ros-jazzy-ros2-controllers

# 1) Make sure Gazebo can find your model folder:
export GZ_SIM_RESOURCE_PATH=$GZ_SIM_RESOURCE_PATH:$(pwd)

# 2) Launch the world (from your package root)
gz sim worlds/arm_world.sdf

# 3) In another terminal, start controllers:
ros2 run controller_manager spawner joint_state_broadcaster
ros2 run controller_manager spawner arm_position_controller

# 4) Drive the joints
ros2 run arm3dof move_demo
# or publish manually:
# ros2 topic pub /arm_position_controller/commands std_msgs/msg/Float64MultiArray "{data: [0.5, 0.3, -0.2]}"
